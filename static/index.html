FastAPI Route Handler

async def get_user(user_id: int):
    user = await db.fetch_one(
        'SELECT * FROM users WHERE id = :id',
        {'id': user_id}
    )
    if not user:
        raise HTTPException(
            status_code=404,
            detail='User not found'
        )
    return user
